import urllib2 name = u'张宏科'

school = u'北京交通大学'

transfer = lambda temp : urllib2.quote(temp.encode('utf-8'))

url = 'https://duckduckgo.com/?q=%s+%s&t=lm' % (transfer(name),transfer(school))
